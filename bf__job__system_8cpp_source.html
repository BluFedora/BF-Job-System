187 template<std::size_t k_MaxTask>
188 struct TaskPool
189 {
190 static_assert(k_MaxTask > 0u, "Must store at least one task in this pool.");
191
192 static constexpr std::size_t k_MaxTaskMinusOne = k_MaxTask - 1u;
193
194 union TaskMemoryBlock
195 {
196 TaskMemoryBlock* next;
197 std::aligned_storage_t<sizeof(Task), alignof(Task)> storage;
198 };
199
200 TaskMemoryBlock memory[k_MaxTask];
201 TaskMemoryBlock* current_block;
202
203 TaskPool()
204 {
205 for (std::size_t i = 0u; i < k_MaxTaskMinusOne; ++i)
206 {
207 memory[i].next = &memory[i + 1u];
208 }
209 memory[k_MaxTaskMinusOne].next = nullptr;
210
211 current_block = &memory[0];
212 }
213
214 Task* allocate(WorkerID worker, TaskFn fn, TaskPtr parent)
215 {
216 TaskMemoryBlock* const result = std::exchange(current_block, current_block->next);
217
218 JobAssert(result != nullptr, "Allocation failure.");
219
220 return new (result) Task(worker, fn, parent);
221 }
222
223 std::size_t indexOf(const Task* const task) const
224 {
225 const TaskMemoryBlock* const block = reinterpret_cast<const TaskMemoryBlock*>(task);
226
227 JobAssert(block >= memory && block < (memory + k_MaxTask), "Invalid task pointer passed in.");
228
229 return block - memory;
230 }
231
232 Task* fromIndex(const std::size_t idx)
233 {
234 JobAssert(idx < k_MaxTask, "Invalid index.");
235
236 return reinterpret_cast<Task*>(&memory[idx].storage);
237 }
238
239 void deallocate(Task* const task)
240 {
241 task->~Task();
242
243 TaskMemoryBlock* const block = reinterpret_cast<TaskMemoryBlock*>(task);
244
245 block->next = std::exchange(current_block, block);
246 }

501 void shutdown() noexcept
502 {
503 JobAssert(s_NextThreadLocalIndex != 0u, "Job System must be initialized before it can be shutdown.");
504
505 s_JobCtx.is_running = false;
506
507 const std::uint32_t num_workers = s_JobCtx.num_workers;
508
509 // Allow one last update loop to allow them to end.
510 s_JobCtx.wakeUpAllWorkers();
511
512 for (std::uint32_t i = 0; i < num_workers; ++i)
513 {
514 ThreadWorker* const worker = s_JobCtx.workers() + i;
515
516 if (i != k_MainThreadID)
517 {
518 worker->stop();
519 }
520
521 worker->~ThreadWorker();
522 }
523
524 s_NextThreadLocalIndex = 0u;
525 s_IsFullyInitialized = false;
526 }

528 Task* taskMake(TaskFn function, Task* const parent) noexcept
529 {
530 const WorkerID worker_id = currentWorker();
531
532 ThreadWorker* const worker = s_JobCtx.workers() + worker_id;
533
534 if (worker->num_allocated_tasks == k_MaxTasksPerWorker)
535 {
536 s_JobCtx.wakeUpAllWorkers();
537
538 // While we cannot allocate do some work.
539 worker->garbageCollectAllocatedTasks();
540 while (worker->num_allocated_tasks == k_MaxTasksPerWorker)
541 {
542 worker->run(worker_id);
543 worker->garbageCollectAllocatedTasks();
544 }
545 }
546
547 JobAssert(worker->num_allocated_tasks < k_MaxTasksPerWorker, "Too many tasks allocated.");
548
549 Task* const task = worker->task_memory.allocate(worker_id, function, parent ? parent->toTaskPtr() : TaskPtr{NullTaskHandle, NullTaskHandle});
550 const TaskHandle task_hdl = TaskHandle(worker->task_memory.indexOf(task));
551
552 if (parent)
553 {
554 parent->num_unfinished_tasks.fetch_add(1u);
555 }
556
557 worker->allocated_tasks[worker->num_allocated_tasks++] = task_hdl;
558 ++task->ref_count;
559
560 return task;
561 }

584 static void taskSubmitQPushHelper(Task* const self, const WorkerID worker_id, QueueType& queue)
585 {

597 Task* taskSubmit(Task* const self, QueueType queue) noexcept
598 {
599 JobAssert(self->q_type == k_InvalidQueueType, "A task cannot be submitted to a queue multiple times.");
600
601 const WorkerID num_workers = numWorkers();
602 const WorkerID worker_id = currentWorker();
603
604 // If we only have one thread running using the worker queue is invalid.
605 if (num_workers == 1u && queue == QueueType::WORKER)
606 {
607 queue = QueueType::NORMAL;
608 }
609
610 ThreadWorker* const worker = s_JobCtx.workers() + worker_id;
611
612 self->q_type = queue;
613
614 switch (queue)
615 {
616 case QueueType::NORMAL:
617 {
618 taskSubmitQPushHelper(self, worker_id, worker->normal_queue);
619 break;
620 }
621 case QueueType::MAIN:
622 {
623 taskSubmitQPushHelper(self, worker_id, s_JobCtx.main_queue);
624 break;
625 }
626 case QueueType::WORKER:
627 {
628 taskSubmitQPushHelper(self, worker_id, worker->worker_queue);
629 break;
630 }
631 default:
632#if defined(__GNUC__) // GCC, Clang, ICC
633 __builtin_unreachable();
634#elif defined(_MSC_VER) // MSVC
635 __assume(false);
636#endif
637 break;
638 }
639
640 if (queue != QueueType::MAIN)
641 {
642 const std::int32_t num_pending_jobs = s_JobCtx.num_available_jobs.fetch_add(1);
643
644 if (num_pending_jobs >= num_workers)
645 {
646 s_JobCtx.wakeUpAllWorkers();
647 }
648 else
649 {
650 s_JobCtx.wakeUpOneWorker();
651 }
652 }
653
654 return self;
655 }

710 void JobSystem::sleep()
711 {
712 if (is_running)
713 {
714 Job::PauseProcessor();
715
716 if (num_available_jobs.load(std::memory_order_relaxed) == 0u)
717 {
718 std::unique_lock<std::mutex> lock(worker_sleep_mutex);
719 worker_sleep_cv.wait(lock, [this]() {
720 // NOTE(SR):
721 // Because the stl wants 'false' to mean continue waiting the logic is a bit confusing :/
722 //
723 // Returns false if the waiting should be continued, aka num_queued_jobs == 0u (also return true if not running).
724 //
725 // Wait If: running AND num_available_jobs == 0.
726 // Do Not Wait If: not running OR num_available_jobs != 0.
727 //
728 return !is_running || num_available_jobs.load(std::memory_order_relaxed) != 0; });
729 }
730 }

754 void Task::onFinish() noexcept
755 {
756 // NOTE(SR):
757 // make sure to store the result in a local variable and use that for
758 // comparing against zero. otherwise, the code contains a data race
759 // because other child tasks could change `num_unfinished_tasks` in the meantime.
760 const std::int32_t num_jobs_left = --num_unfinished_tasks;
761
762 if (num_jobs_left == 0)
763 {
764 if (!parent.isNull())
765 {
766 Task* const parent_task = taskPtrToPointer(parent);
767
768 parent_task->onFinish();
769 }
770
771 --num_unfinished_tasks;
772
773 TaskPtr continuation = first_continuation.load();
774
775 while (!continuation.isNull())

903 bool ThreadWorker::run(const WorkerID worker_id)
904 {
905 const bool is_main_thread = worker_id == k_MainThreadID;
906
907 TaskPtr task_ptr = normal_queue.pop();
908
909 if (task_ptr.isNull())
910 {
911 task_ptr = is_main_thread ? s_JobCtx.main_queue.pop() : worker_queue.pop();
912 }
913
914 if (task_ptr.isNull())
915 {
916 const WorkerID other_worker_id = randomWorker();
917
918 if (other_worker_id != worker_id)
919 {
920 ThreadWorker* const other_worker = s_JobCtx.workers() + other_worker_id;
921
922 task_ptr = other_worker->normal_queue.steal();
923
924 if (task_ptr.isNull() && !is_main_thread)
925 {
926 task_ptr = other_worker->worker_queue.steal();
927 }
928 }
929 }
930
931 if (task_ptr.isNull())
932 {
933 return false;
934 }
935
936 s_JobCtx.num_available_jobs.fetch_sub(1);
937
938 Task* const task = taskPtrToPointer(task_ptr);
939 task->run();
940
941 return true;
942 }
