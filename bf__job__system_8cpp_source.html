688  void ThreadWorker::garbageCollectAllocatedTasks(const WorkerID thread_id)
689  {
690  if (num_allocated_tasks)
691  {
692  const auto allocated_tasks_bgn = allocated_tasks.begin();
693  const auto allocated_tasks_end = allocated_tasks_bgn + num_allocated_tasks;
694 
695  // NOTE(SR):
696  // Cannot use `remove_if` since it moves the removed elements,
697  // but we want swaps so that we can safely read the removed
698  // elements from done_end => allocated_tasks_end. (aka that next for_each loop)
699  const auto done_end = std::partition(
700  allocated_tasks_bgn,
701  allocated_tasks_end,
702  [thread_id](TaskHandle task_hdl) {
703  Task* const task = taskPtrToPointer({thread_id, task_hdl});
704 
705  return !taskIsDone(task);
706  });
707 
708  std::for_each(
709  done_end,
710  allocated_tasks_end,
711  [&task_memory = task_memory, thread_id](TaskHandle task_hdl) {
712  Task* const task = taskPtrToPointer({thread_id, task_hdl});
713 
714  task_memory.deallocateT(task);
715  });
716 
717  num_allocated_tasks = TaskHandle(std::distance(allocated_tasks_bgn, done_end));
718  }
