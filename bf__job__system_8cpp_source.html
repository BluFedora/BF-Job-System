184 template<std::size_t k_MaxTask>
185 struct TaskPool
186 {
187 static_assert(k_MaxTask > 0u, "Must store at least one task in this pool.");
188
189 static constexpr std::size_t k_MaxTaskMinusOne = k_MaxTask - 1u;
190
191 union TaskMemoryBlock
192 {
193 TaskMemoryBlock* next;
194 std::aligned_storage_t<sizeof(Task), alignof(Task)> storage;
195 };
196
197 TaskMemoryBlock memory[k_MaxTask];
198 TaskMemoryBlock* current_block;
199
200 TaskPool()
201 {
202 for (std::size_t i = 0u; i < k_MaxTaskMinusOne; ++i)
203 {
204 memory[i].next = &memory[i + 1u];
205 }
206 memory[k_MaxTaskMinusOne].next = nullptr;
207
208 current_block = &memory[0];
209 }
210
211 Task* allocate(WorkerID worker, TaskFn fn, TaskPtr parent)
212 {
213 TaskMemoryBlock* const result = std::exchange(current_block, current_block->next);
214
215 JobAssert(result != nullptr, "Allocation failure.");
216
217 return new (result) Task(worker, fn, parent);
218 }
219
220 std::size_t indexOf(const Task* const task) const
221 {
222 const TaskMemoryBlock* const block = reinterpret_cast<const TaskMemoryBlock*>(task);
223
224 JobAssert(block >= memory && block < (memory + k_MaxTask), "Invalid task pointer passed in.");
225
226 return block - memory;
227 }
228
229 Task* fromIndex(const std::size_t idx)
230 {
231 JobAssert(idx < k_MaxTask, "Invalid index.");
232
233 return reinterpret_cast<Task*>(&memory[idx].storage);
234 }
235
236 void deallocate(Task* const task)
237 {
238 task->~Task();
239
240 TaskMemoryBlock* const block = reinterpret_cast<TaskMemoryBlock*>(task);
241
242 block->next = std::exchange(current_block, block);
243 }

492 void shutdown() noexcept
493 {
494 JobAssert(s_NextThreadLocalIndex != 0u, "Job System must be initialized before it can be shutdown.");
495
496 s_JobCtx.is_running = false;
497
498 const std::uint32_t num_workers = s_JobCtx.num_workers;
499
500 // Allow one last update loop to allow them to end.
501 s_JobCtx.wakeUpAllWorkers();
502
503 for (std::uint32_t i = 0; i < num_workers; ++i)
504 {
505 ThreadWorker* const worker = s_JobCtx.workers() + i;
506
507 if (i != k_MainThreadID)

512 worker->~ThreadWorker();
513 }

515 s_NextThreadLocalIndex = 0u;
516 s_IsFullyInitialized = false;

519 Task* taskMake(TaskFn function, Task* const parent) noexcept
520 {
521 const WorkerID worker_id = currentWorker();
522
523 ThreadWorker* const worker = s_JobCtx.workers() + worker_id;
524
525 if (worker->num_allocated_tasks == k_MaxTasksPerWorker)
526 {
527 s_JobCtx.wakeUpAllWorkers();
528
529 // While we cannot allocate do some work.
530 worker->garbageCollectAllocatedTasks();
531 while (worker->num_allocated_tasks == k_MaxTasksPerWorker)
532 {
533 worker->run(worker_id);
534 worker->garbageCollectAllocatedTasks();
535 }
536 }
537
538 JobAssert(worker->num_allocated_tasks < k_MaxTasksPerWorker, "Too many tasks allocated.");
539
540 Task* const task = worker->task_memory.allocate(worker_id, function, parent ? parent->toTaskPtr() : TaskPtr{NullTaskHandle, NullTaskHandle});
541 const TaskHandle task_hdl = TaskHandle(worker->task_memory.indexOf(task));
542
543 if (parent)
544 {
545 parent->num_unfinished_tasks.fetch_add(1u);
546 }
547
548 worker->allocated_tasks[worker->num_allocated_tasks++] = task_hdl;
549 ++task->ref_count;
550
551 return task;
552 }

576 template<typename QueueType>
577 static void taskSubmitQPushHelper(Task* const self, const WorkerID worker_id, QueueType& queue)
578 {
579 ThreadWorker* const worker = s_JobCtx.workers() + worker_id;
580 const TaskPtr task_ptr = self->toTaskPtr();
581
582 // Loop until we have successfully pushed to the queue.

590 Task* taskSubmit(Task* const self, QueueType queue) noexcept
591 {
592 const WorkerID num_workers = numWorkers();
593 const WorkerID worker_id = currentWorker();
594
595 // If we only have one thread running using the working queue is invalid.
596 if (num_workers == 1u && queue == QueueType::WORKER)
597 {
598 queue = QueueType::NORMAL;
599 }
600
601 JobAssert(self->q_type == k_InvalidQueueType, "A task cannot be submitted to a queue multiple times.");
602
603 ThreadWorker* const worker = s_JobCtx.workers() + worker_id;
604
605 self->q_type = queue;
606
607 switch (queue)
608 {
609 case QueueType::NORMAL:
610 {
611 taskSubmitQPushHelper(self, worker_id, worker->hi_queue);
612 break;
613 }
614 case QueueType::MAIN:
615 {
616 taskSubmitQPushHelper(self, worker_id, s_JobCtx.main_queue);
617 break;
618 }
619 case QueueType::WORKER:
620 {
621 taskSubmitQPushHelper(self, worker_id, worker->bg_queue);
622 break;
623 }
624 }
625
626 if (queue != QueueType::MAIN)
627 {
628 const std::int32_t num_pending_jobs = s_JobCtx.num_available_jobs.fetch_add(1);
629
630 if (num_pending_jobs >= num_workers)
631 {
632 s_JobCtx.wakeUpAllWorkers();
633 }
634 else
635 {
636 s_JobCtx.wakeUpOneWorker();
637 }
638 }
639
640 return self;
641 }

691 void JobSystem::sleep(const ThreadWorker* const worker)
692 {
693 if (is_running)
694 {
695 std::this_thread::yield();
696
697 if (num_available_jobs.load(std::memory_order_relaxed) == 0u)
698 {
699 std::unique_lock<std::mutex> lock(worker_sleep_mutex);
700 worker_sleep_cv.wait(lock, [worker, this]() {
701 // NOTE(SR):
702 // Because the stl wants 'false' to mean continue waiting the logic is a bit confusing :/
703 //
704 // Returns false if the waiting should be continued, aka num_queued_jobs == 0u (also return true if not running).
705 //
706 // Wait If: running AND num_available_jobs == 0.
707 // Do Not Wait If: not running OR num_available_jobs != 0.
708 //
709 return !is_running || num_available_jobs.load(std::memory_order_relaxed) != 0;
710 });
711 }
712 }

736 void Task::onFinish() noexcept
737 {
738 // NOTE(SR):
739 // make sure to store the result in a local variable and use that for
740 // comparing against zero. otherwise, the code contains a data race
741 // because other child tasks could change `num_unfinished_tasks` in the meantime.
742 const std::int32_t num_jobs_left = --num_unfinished_tasks;
743
744 if (num_jobs_left == 0)
745 {
746 if (!parent.isNull())
747 {
748 Task* const parent_task = taskPtrToPointer(parent);
749
750 parent_task->onFinish();
751 }
752
753 --num_unfinished_tasks;
754
755 TaskPtr continuation = first_continuation.load();
756
757 while (!continuation.isNull())
758 {
759 Task* const task = taskPtrToPointer(continuation);
760 const TaskPtr next_continuation = task->next_continuation;
761
762 taskSubmit(task, q_type);
763
764 continuation = next_continuation;
765 }
766
767 --ref_count;
768 }

771 void ThreadWorker::start(const WorkerID worker_index, const std::uint64_t rng_seed)
772 {
773 const bool is_main_thread = worker_index == k_MainThreadID;
774
775 pcg32_srandom_r(&rng_state, std::uint64_t(worker_index) + rng_seed, std::uint64_t(worker_index) * 2u + 1u + rng_seed);
776
777 if (is_main_thread)
778 {
779 initThreadWorkerID();
780 }
781 else
782 {
783 new (worker()) std::thread([]() {
784 initThreadWorkerID();

885 bool ThreadWorker::run(const WorkerID worker_id)
886 {
887 const bool is_main_thread = worker_id == k_MainThreadID;
888
889 TaskPtr task_ptr = hi_queue.pop();
890
891 if (task_ptr.isNull())
892 {
893 task_ptr = is_main_thread ? s_JobCtx.main_queue.pop() : bg_queue.pop();
894 }
895
896 if (task_ptr.isNull())
897 {
898 const WorkerID other_worker_id = randomWorker();
899
900 if (other_worker_id != worker_id)
901 {
902 ThreadWorker* const other_worker = s_JobCtx.workers() + other_worker_id;
903
904 task_ptr = other_worker->hi_queue.steal();
905
906 if (task_ptr.isNull() && !is_main_thread)
907 {
908 task_ptr = other_worker->bg_queue.steal();
909 }
910 }
911 }
912
913 if (task_ptr.isNull())
914 {
915 return false;
916 }
917
918 s_JobCtx.num_available_jobs.fetch_sub(1);
919
920 Task* const task = taskPtrToPointer(task_ptr);
921 task->run();
922
923 return true;
924 }

933 WorkerID ThreadWorker::randomWorker() noexcept
934 {
935 const std::uint32_t num_workers = s_JobCtx.num_workers;
936
937 std::uint32_t weights[k_MaxThreadsSupported];
938 std::uint32_t weight_sum = 0u;
939
940 for (WorkerID i = 0u; i < num_workers; ++i)
941 {
942 const ThreadWorker& worker = s_JobCtx.workers()[i];
943
944 weights[i] = worker.num_allocated_tasks;
945 weight_sum += weights[i];
946 }
947
948 if (weight_sum != 0)
949 {
950 std::uint32_t random_number = pcg32_boundedrand_r(&rng_state, weight_sum);
951
952 for (WorkerID i = 0u; i < num_workers; ++i)
953 {
954 if (random_number < weights[i])
955 {
956 return i;
957 }
958
959 random_number -= weights[i];
960 }
961 }
962
963 return WorkerID(pcg32_boundedrand_r(&rng_state, num_workers));
964 }
