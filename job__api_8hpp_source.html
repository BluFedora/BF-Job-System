621 template<typename F, typename S>
622 Task* ParallelFor(const std::size_t start, const std::size_t count, S&& splitter, F&& fn, Task* parent = nullptr)
623 {
624 return TaskMake(
625 [=, splitter = std::move(splitter), fn = std::move(fn)](Task* const task) {
626 if (count > 1u && splitter(count))
627 {
628 const std::size_t left_count = count / 2;
629 const std::size_t right_count = count - left_count;
630 const QueueType parent_q_type = detail::taskQType(task);
631
632 TaskSubmit(ParallelFor(start, left_count, splitter, fn, task), parent_q_type);
633 TaskSubmit(ParallelFor(start + left_count, right_count, splitter, fn, task), parent_q_type);
634 }
635 else
636 {
637 fn(task, IndexRange{start, start + count});
638 }
639 },
640 parent);
641 }

643 template<typename Splitter, typename Reducer>
644 Task* ParallelReduce(const std::size_t start, const std::size_t count, Splitter&& splitter, Reducer&& reduce, Task* parent = nullptr)
645 {
646 return TaskMake(
647 [=, splitter = std::move(splitter), reduce = std::move(reduce)](Task* const task) {
648 // NOTE(SR):
649 // Could also have a stride that increases each step.
650 // This would be bad for Cuda GPU (Shared Memory Bank Conflict)
651 // But good on CPU with better locality.
652 // https://developer.download.nvidia.com/assets/cuda/files/reduction.pdf
653
654 const QueueType parent_q_type = detail::taskQType(task);
655 std::size_t count_left = count;
656 while (count_left > 1)
657 {
658 const std::size_t stride = count_left / 2;
659
660 const auto ReduceRange = [stride, &reduce](Task* const sub_task, const IndexRange range) {
661 for (const std::size_t index : range)
662 {
663 reduce(sub_task, index, index + stride);
664 }
665 };
666
667 TaskSubmitAndWait(ParallelFor(start, stride, splitter, ReduceRange, nullptr), parent_q_type);
668
669 if ((count_left & 1) != 0)
670 {
671 reduce(task, start, start + count_left - 1);
672 }
673
674 count_left = stride;
675 }
676 },
677 parent);
678 }
