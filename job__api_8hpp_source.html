497 Task* TaskMake(Closure&& function, Task* const parent)
498 {
499 Task* const task = TaskMake(
500 +[](Task* const task) -> void {
501 Closure& function = *static_cast<Closure*>(detail::taskGetPrivateUserData(task, alignof(Closure)));
502 function(task);
503 function.~Closure();
504 },
505 parent);
506 void* const private_data = detail::taskReservePrivateUserData(task, sizeof(Closure), alignof(Closure));
507 new (private_data) Closure(std::forward<Closure>(function));

625 template<typename F, typename S>
626 Task* ParallelFor(const std::size_t start, const std::size_t count, S&& splitter, F&& fn, Task* parent = nullptr)
627 {
628 return TaskMake(
629 [=, splitter = std::move(splitter), fn = std::move(fn)](Task* const task) {
630 if (count > 1u && splitter(count))
631 {
632 const std::size_t left_count = count / 2;
633 const std::size_t right_count = count - left_count;
634 const QueueType parent_q_type = detail::taskQType(task);
635
636 TaskSubmit(ParallelFor(start, left_count, splitter, fn, task), parent_q_type);
637 TaskSubmit(ParallelFor(start + left_count, right_count, splitter, fn, task), parent_q_type);
638 }
639 else
640 {
641 fn(task, IndexRange{start, start + count});
642 }
643 },
644 parent);
645 }

647 template<typename Splitter, typename Reducer>
648 Task* ParallelReduce(const std::size_t start, const std::size_t count, Splitter&& splitter, Reducer&& reduce, Task* parent = nullptr)
649 {
650 return TaskMake(
651 [=, splitter = std::move(splitter), reduce = std::move(reduce)](Task* const task) {
652 // NOTE(SR):
653 // Could also have a stride that increases each step.
654 // This would be bad for Cuda GPU (Shared Memory Bank Conflict)
655 // But good on CPU with better locality.
656 // https://developer.download.nvidia.com/assets/cuda/files/reduction.pdf
657
658 const QueueType parent_q_type = detail::taskQType(task);
659 std::size_t count_left = count;
660 while (count_left > 1)
661 {
662 const std::size_t stride = count_left / 2;
663
664 const auto ReduceRange = [stride, &reduce](Task* const sub_task, const IndexRange range) {
665 for (const std::size_t index : range)
666 {
667 reduce(sub_task, index, index + stride);
668 }
669 };
670
671 TaskSubmitAndWait(ParallelFor(start, stride, splitter, ReduceRange, nullptr), parent_q_type);
672
673 if ((count_left & 1) != 0)
674 {
675 reduce(task, start, start + count_left - 1);
676 }
677
678 count_left = stride;
679 }
680 },
681 parent);
682 }
