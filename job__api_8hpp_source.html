616 template<typename F, typename S>
617 Task* ParallelFor(const std::size_t start, const std::size_t count, S&& splitter, F&& fn, Task* parent = nullptr)
618 {
619 return TaskMake(
620 [=, splitter = std::move(splitter), fn = std::move(fn)](Task* const task) {
621 if (count > 1u && splitter(count))
622 {
623 const std::size_t left_count = count / 2;
624 const std::size_t right_count = count - left_count;
625 const QueueType parent_q_type = detail::taskQType(task);
626
627 if (left_count)
628 {
629 TaskSubmit(ParallelFor(start, left_count, splitter, fn, task), parent_q_type);
630 }
631
632 if (right_count)
633 {
634 TaskSubmit(ParallelFor(start + left_count, right_count, splitter, fn, task), parent_q_type);
635 }
636 }
637 else
638 {
639 fn(task, IndexRange{start, start + count});
640 }
641 },
642 parent);
643 }
