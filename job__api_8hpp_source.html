582 template<typename F, typename S>
583 Task* ParallelFor(const std::size_t start, const std::size_t count, S&& splitter, F&& fn, Task* parent = nullptr)
584 {
585 return TaskMake(
586 [=, splitter = std::move(splitter), fn = std::move(fn)](Task* const task) {
587 if (count > 1u && splitter(count))
588 {
589 const std::size_t left_count = count / 2;
590 const std::size_t right_count = count - left_count;
591 const QueueType parent_q_type = detail::taskQType(task);
592
593 if (left_count)
594 {
595 TaskSubmit(ParallelFor(start, left_count, splitter, fn, task), parent_q_type);
596 }
597
598 if (right_count)
599 {
600 TaskSubmit(ParallelFor(start + left_count, right_count, splitter, fn, task), parent_q_type);
601 }
602 }
603 else
604 {
605 fn(task, IndexRange{start, start + count});
606 }
607 },
608 parent);
609 }
