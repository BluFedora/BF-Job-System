33 template<typename T>
34 class LockedQueue
35 {
36 public:
37 using size_type = std::size_t;
38
39 private:
40 std::mutex m_Lock;
41 T* m_Data;
42 size_type m_Capacity;
43 size_type m_CapacityMask;
44 size_type m_WriteIndex;
45 size_type m_Size;
46
47 public:
48 void Initialize(T* const memory_backing, const size_type capacity) noexcept
49 {
50 m_Data = memory_backing;
51 m_Capacity = capacity;
52 m_CapacityMask = capacity - 1;
53 m_WriteIndex = 0u;
54 m_Size = 0u;
55
56 JobAssert((m_Capacity & m_CapacityMask) == 0, "Capacity must be a power of 2.");
57 }
58
59 bool Push(const T& value)
60 {
61 const std::lock_guard<std::mutex> guard(m_Lock);
62 (void)guard;
63
64 if (m_Size == m_Capacity)
65 {
66 return false;
67 }
68
69 *elementAt(m_WriteIndex++) = value;
70 ++m_Size;
71
72 return true;
73 }
74
75 bool Pop(T* const out_value)
76 {
77 JobAssert(out_value != nullptr, "`out_value` cannot be a nullptr.");
78
79 const std::lock_guard<std::mutex> guard(m_Lock);
80 (void)guard;
81
82 if (m_Size == 0u)
83 {
84 return false;
85 }
86
87 *out_value = *elementAt(m_WriteIndex - m_Size);
88 --m_Size;
89
90 return true;
91 }
92
93 private:
94 size_type mask(const size_type raw_index) const noexcept
95 {
96 return raw_index & m_CapacityMask;
97 }
98
99 T* elementAt(const size_type raw_index) const noexcept
100 {
101 return m_Data + mask(raw_index);
102 }
103 };

110 class SPSCQueue
111 {
112 public:
113 using size_type = std::size_t;
114 using atomic_size_type = std::atomic<size_type>;
115
116 private:
117 // Writer Thread
118
119 alignas(k_FalseSharingPadSize) atomic_size_type m_ProducerIndex;
120 unsigned char m_Padding0[k_FalseSharingPadSize - sizeof(m_ProducerIndex)];
121 alignas(k_FalseSharingPadSize) size_type m_CachedConsumerIndex;
122 unsigned char m_Padding1[k_FalseSharingPadSize - sizeof(m_CachedConsumerIndex)];
123
124 // Reader Thread
125
126 alignas(k_FalseSharingPadSize) atomic_size_type m_ConsumerIndex;
127 unsigned char m_Padding2[k_FalseSharingPadSize - sizeof(m_ConsumerIndex)];
128 alignas(k_FalseSharingPadSize) size_type m_CachedProducerIndex;
129 unsigned char m_Padding3[k_FalseSharingPadSize - sizeof(m_CachedProducerIndex)];
130
131 // Shared 'Immutable' State
132
133 alignas(k_FalseSharingPadSize) T* m_Data;
134 size_type m_Capacity;
135 size_type m_CapacityMask;

137 static_assert(atomic_size_type::is_always_lock_free, "Expected to be lockfree.");
138

143 // NOTE(SR): Not thread safe.
144 void Initialize(T* const memory_backing, const size_type capacity) noexcept
145 {
146 m_ProducerIndex.store(0, std::memory_order_relaxed);
147 m_CachedConsumerIndex = 0;
148 m_ConsumerIndex.store(0, std::memory_order_relaxed);
149 m_CachedProducerIndex = 0;
150 m_Data = memory_backing;
151 m_Capacity = capacity;
152 m_CapacityMask = capacity - 1;
153
154 JobAssert((m_Capacity & m_CapacityMask) == 0, "Capacity must be a power of 2.");
155 }
156
157 bool Push(const T& value)
158 {
159 return PushLazy([&value](T* const destination) { *destination = value; });
160 }
161
162 bool Pop(T* const out_value)
163 {
164 JobAssert(out_value != nullptr, "`out_value` cannot be a nullptr.");
165
166 return PopLazy([out_value](T&& value) { *out_value = std::move(value); });
167 }
168
169 // Memory passed into `callback` is uninitialized, must be placement new'ed into.
170 template<typename CallbackFn>
171 bool PushLazy(CallbackFn&& callback)
172 {
173 const size_type write_index = m_ProducerIndex.load(std::memory_order_relaxed);
174
175 if (IsFull(write_index, m_CachedConsumerIndex))
176 {
177 m_CachedConsumerIndex = m_ConsumerIndex.load(std::memory_order_acquire);
178 if (IsFull(write_index, m_CachedConsumerIndex))
179 {
180 return false;
181 }
182 }
183
184 callback(ElementAt(write_index));
185 m_ProducerIndex.store(write_index + 1, std::memory_order_release);
186
187 return true;
188 }
189
190 template<typename CallbackFn>
191 bool PopLazy(CallbackFn&& callback)
192 {
193 const size_type read_index = m_ConsumerIndex.load(std::memory_order_relaxed);
194
195 if (IsEmpty(m_CachedProducerIndex, read_index))
196 {
197 m_CachedProducerIndex = m_ProducerIndex.load(std::memory_order_acquire);
198 if (IsEmpty(m_CachedProducerIndex, read_index))
199 {
200 return false;
201 }
202 }
203
204 T* const element = ElementAt(read_index);
205 callback(std::move(*element));
206 element->~T();
207 m_ConsumerIndex.fetch_add(1, std::memory_order_release);
208
209 return true;
210 }
211
212 private:
213 bool IsFull(const size_type head, const size_type tail) const noexcept
214 {
215 return ((head + 1) & m_CapacityMask) == tail;
216 }
217
218 static bool IsEmpty(const size_type head, const size_type tail) noexcept
219 {
220 return head == tail;
221 }
222
223 T* ElementAt(const size_type index) const noexcept
224 {
225 return m_Data + (index & m_CapacityMask);
226 }

238 template<typename T>
239 class SPMCDeque
240 {
241 private:
242 using AtomicT = std::atomic<T>;
243
244 static_assert(AtomicT::is_always_lock_free, "T Should be a small pointer-like type, expected to be lock-free when atomic.");
245
246 public:
247 using size_type = std::int64_t; // NOTE(SR): Must be signed for Pop to work correctly on empty queue.
248 using atomic_size_type = std::atomic<size_type>;
249
250 private:
251 atomic_size_type m_ProducerIndex;
252 atomic_size_type m_ConsumerIndex;
253 unsigned char m_Padding0[k_FalseSharingPadSize - sizeof(atomic_size_type) * 2];
254
255 // Shared 'Immutable' State
256
257 alignas(k_FalseSharingPadSize) AtomicT* m_Data;
258 size_type m_Capacity;
259 size_type m_CapacityMask;
260
261 public:
262 SPMCDeque() = default;

265 // NOTE(SR): Not thread safe.
266 void Initialize(AtomicT* const memory_backing, const size_type capacity) noexcept
267 {
268 m_ProducerIndex = 0;
269 m_ConsumerIndex = 0;
270 m_Data = memory_backing;
271 m_Capacity = capacity;
272 m_CapacityMask = capacity - 1;
273
274 JobAssert((m_Capacity & m_CapacityMask) == 0, "Capacity must be a power of 2.");
275 }
276
277 // NOTE(SR): Must be called by owning thread.
278
279 SPMCDequeStatus Push(const T& value)
280 {
281 const size_type write_index = m_ProducerIndex.load(std::memory_order_relaxed);
282 const size_type read_index = m_ConsumerIndex.load(std::memory_order_acquire);
283 const size_type size = write_index - read_index;
284
285 if (size > m_CapacityMask)
286 {
287 return SPMCDequeStatus::FAILED_SIZE;
288 }
289
290 ElementAt(write_index)->store(value, std::memory_order_relaxed);
291
292 m_ProducerIndex.store(write_index + 1, std::memory_order_release);
293
294 return SPMCDequeStatus::SUCCESS;
295 }
296
297 SPMCDequeStatus Pop(T* const out_value)
298 {
299 const size_type producer_index = m_ProducerIndex.load(std::memory_order_relaxed) - 1;
300
301 // Reserve the slot at the producer end.
302 m_ProducerIndex.store(producer_index, std::memory_order_relaxed);
303
304 // The above store needs to happen before this next read
305 // to have consistent view of the buffer.
306 //
307 // `m_ProducerIndex` can only be written to by this thread
308 // so first reserve a slot then we read what the other threads have to say.
309 //
310 std::atomic_thread_fence(std::memory_order_seq_cst);
311
312 size_type consumer_index = m_ConsumerIndex.load(std::memory_order_relaxed);
313
314 if (consumer_index <= producer_index)
315 {
316 if (consumer_index == producer_index) // Only one item in queue
317 {
318 const bool successful_pop = m_ConsumerIndex.compare_exchange_strong(consumer_index, consumer_index + 1, std::memory_order_seq_cst, std::memory_order_relaxed);
319
320 if (successful_pop)
321 {
322 *out_value = ElementAt(producer_index)->load(std::memory_order_relaxed);
323 }
324
325 m_ProducerIndex.store(producer_index + 1, std::memory_order_relaxed);
326 return successful_pop ? SPMCDequeStatus::SUCCESS : SPMCDequeStatus::FAILED_RACE;
327 }
328
329 *out_value = ElementAt(producer_index)->load(std::memory_order_relaxed);
330 return SPMCDequeStatus::SUCCESS;
331 }
332
333 // Empty Queue, so restore to canonical empty.
334 m_ProducerIndex.store(producer_index + 1, std::memory_order_seq_cst);
335 return SPMCDequeStatus::FAILED_SIZE;
336 }
337
338 // NOTE(SR): Must be called by non owning thread.
339
340 SPMCDequeStatus Steal(T* const out_value)
341 {
342 size_type read_index = m_ConsumerIndex.load(std::memory_order_acquire);
343
344 // Must fully read `m_ConsumerIndex` before we read the producer owned `m_ProducerIndex`.
345 std::atomic_thread_fence(std::memory_order_seq_cst);
346
347 const size_type write_index = m_ProducerIndex.load(std::memory_order_acquire);
348
349 // if (next_read_index <= write_index)
350 if (read_index < write_index)
351 {
352 // Must load result before the CAS, since a push can happen concurrently right after the CAS.
353 T result = ElementAt(read_index)->load(std::memory_order_relaxed);
354
355 // Need strong memory ordering to read the element before the cas.
356 if (m_ConsumerIndex.compare_exchange_strong(read_index, read_index + 1, std::memory_order_seq_cst, std::memory_order_relaxed))
357 {
358 *out_value = std::move(result);
359 return SPMCDequeStatus::SUCCESS;
360 }
361
362 return SPMCDequeStatus::FAILED_RACE;
363 }
364
365 return SPMCDequeStatus::FAILED_SIZE;
366 }
367
368 private:
369 AtomicT* ElementAt(const size_type index) const noexcept
370 {
371 return m_Data + (index & m_CapacityMask);
372 }

441 size_type PushImpl(const value_type* elements, const size_type num_elements)
442 {

444 if (RequestWriteRange<allOrNothing>(&range, num_elements))
445 {

457 IndexRange range;
458 if (RequestPopRange<allOrNothing>(&range, num_elements))
459 {
460 const size_type read_elements = ReadElements(out_elements, range);
461 Commit(&m_ConsumerCommited, range);

468 template<bool allOrNothing>
469 bool RequestWriteRange(IndexRange* out_range, const size_type num_items)
470 {
471 size_type old_head, new_head;
472
473 old_head = m_ProducerPending.load(std::memory_order_relaxed);
474 do
475 {
476 const size_type tail = m_ConsumerCommited.load(std::memory_order_acquire);
477 const size_type capacity_left = Distance(old_head, tail);
478 size_type num_element_to_write = num_items;
479
480 if constexpr (allOrNothing)
481 {
482 if (capacity_left <= num_items)
483 {
484 return false;
485 }
486 }
487 else
488 {
489 if (!capacity_left)
490 {
491 return false;
492 }
493
494 if (capacity_left < num_element_to_write)
495 {
496 num_element_to_write = capacity_left;
497 }
498 }
499
500 new_head = old_head + num_element_to_write;
501
502 } while (!m_ProducerPending.compare_exchange_weak(old_head, new_head, std::memory_order_relaxed, std::memory_order_relaxed));
503
504 *out_range = {old_head, new_head};
505 return true;
506 }
507
508 template<bool allOrNothing>
509 bool RequestPopRange(IndexRange* out_range, const size_type num_items)
510 {
511 size_type old_tail, new_tail;
512
513 old_tail = m_ConsumerPending.load(std::memory_order_relaxed);
514 do
515 {
516 const size_type head = m_ProducerCommited.load(std::memory_order_acquire);
517 const size_type distance = Distance(head, old_tail);
518 const size_t capacity_left = (m_Capacity - distance);
519 size_type num_element_to_read = num_items;
520
521 if constexpr (allOrNothing)
522 {
523 if (capacity_left < num_items)
524 {
525 return false;
526 }
527 }
528 else
529 {
530 if (!capacity_left)
531 {
532 return false;
533 }
534
535 if (capacity_left < num_element_to_read)
536 {
537 num_element_to_read = capacity_left;
538 }
539 }
540
541 new_tail = old_tail + num_items;
542
543 } while (!m_ConsumerPending.compare_exchange_weak(old_tail, new_tail, std::memory_order_relaxed, std::memory_order_relaxed));
544
545 *out_range = {old_tail, new_tail};
546 return true;
547 }

549 size_type WriteElements(const value_type* const elements, const IndexRange range)
550 {
551 const size_type real_start = range.start % m_Capacity;
552 const size_type write_size = Distance(real_start, range.end % m_Capacity);
553 const size_type capacity_before_split = m_Capacity - real_start;
554 const size_type num_items_before_split = write_size < capacity_before_split ? write_size : capacity_before_split;
555 const size_type num_items_after_split = write_size - num_items_before_split;
556
557 std::copy_n(elements + 0u, num_items_before_split, m_Queue + real_start);
558 std::copy_n(elements + num_items_before_split, num_items_after_split, m_Queue + 0u);
559
560 return write_size;
561 }
562
563 size_type ReadElements(value_type* const out_elements, const IndexRange range) const
564 {
565 const size_type real_start = range.start % m_Capacity;
566 const size_type read_size = Distance(real_start, range.end % m_Capacity);
567 const size_type capacity_before_split = m_Capacity - real_start;
568 const size_type num_items_before_split = read_size < capacity_before_split ? read_size : capacity_before_split;
569 const size_type num_items_after_split = read_size - num_items_before_split;
570
571 std::copy_n(std::make_move_iterator(m_Queue + real_start), num_items_before_split, out_elements + 0u);
572 std::copy_n(std::make_move_iterator(m_Queue + 0u), num_items_after_split, out_elements + num_items_before_split);
573
574 return read_size;
575 }

577 void Commit(atomic_size_type* commit, const IndexRange range) const
578 {
579 size_type start_copy;
580 while (!commit->compare_exchange_weak(
581 start_copy = range.start,
582 range.end,
583 std::memory_order_release,
584 std::memory_order_relaxed))
585 {
586 PauseProcessor();
587 }
588 }
589
590 size_type Distance(const size_type a, const size_type b) const
591 {
592 return (b > a) ? (b - a) : m_Capacity - a + b;
593 }
