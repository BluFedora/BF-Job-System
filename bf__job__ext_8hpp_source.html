186  template<typename T, typename F, typename S>
187  void parallel_for_impl(Task* task)
188  {
189  using ParallelForDataType = ParallelForData<T, S, F>;
190 
191  ParallelForDataType& data = taskDataAs<ParallelForDataType>(task);
192  F& custom_f = data.job();
193  const std::size_t data_count = data.count();
194 
195  if (data.splitter()(data_count))
196  {
197  const std::size_t left_count = data_count / 2;
198  const std::size_t right_count = data_count - left_count;
199  const QueueType parent_q_type = taskQType(task);
200 
201  if (left_count)
202  {
203  Task* const left_task = taskMake(&detail::parallel_for_impl<T, F, S>, task);
204 
205  taskEmplaceData<ParallelForDataType>(left_task, data.data(), custom_f, left_count, data.splitter());
206 
207  taskSubmit(left_task, parent_q_type);
208  }
209 
210  if (right_count)
211  {
212  Task* const right_task = taskMake(&detail::parallel_for_impl<T, F, S>, task);
213 
214  taskEmplaceData<ParallelForDataType>(right_task, data.data() + left_count, custom_f, right_count, data.splitter());
215 
216  taskSubmit(right_task, parent_q_type);
217  }
218  }
219  else
220  {
221  custom_f(data.data(), data.count());
222  }
