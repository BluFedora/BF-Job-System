127 template<typename F, typename S>
128 Task* parallel_for(const std::size_t start, const std::size_t count, S&& splitter, F&& fn, Task* parent = nullptr)
129 {
130 return taskMake(
131 [=, splitter = std::move(splitter), fn = std::move(fn)](Task* const task) {
132 if (count > 1u && splitter(count))
133 {
134 const std::size_t left_count = count / 2;
135 const std::size_t right_count = count - left_count;
136 const QueueType parent_q_type = detail::taskQType(task);
137
138 if (left_count)
139 {
140 taskSubmit(parallel_for(start, left_count, splitter, fn, task), parent_q_type);
141 }
142
143 if (right_count)
144 {
145 taskSubmit(parallel_for(start + left_count, right_count, splitter, fn, task), parent_q_type);
146 }
147 }
148 else
149 {
150 fn(task, IndexRange{start, start + count});
151 }
152 },
153 parent);
154 }
